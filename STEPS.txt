GST Amount = (Original Price * GST Rate) / 100
Total Amount = Original Price + GST Amount
// GST Calculation Function
function calculateGST(originalPrice, gstRate) {
    // Calculate GST Amount
    const gstAmount = (originalPrice * gstRate) / 100;

    // Calculate Total Amount (inclusive of GST)
    const totalAmount = originalPrice + gstAmount;

    // Return an object with both amounts
    return {
        originalPrice,
        gstAmount,
        totalAmount,
    };
}

// Example Usage
const originalPrice = 1000; // Replace with your actual price
const gstRate = 18; // Replace with your applicable GST rate

const result = calculateGST(originalPrice, gstRate);

console.log("Original Price: Rs.", result.originalPrice);
console.log("GST Amount: Rs.", result.gstAmount);
console.log("Total Amount (inclusive of GST): Rs.", result.totalAmount);
